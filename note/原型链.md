### 原型链
#### 创建对象有几种方法
``` javascript
// 第一种方式：字面量
var o1 = {name: 'o1'}; // Objecty {name:"o1"}
var o2 = new Object({name:'o2'}); // Objecty {name:"o2"}
// 第二种方式：通过构造函数
var M = function(name){this.name = name};
var o3 = new M('o3'); // M {name:"o3"}
// 第三种方式：Object.create
var P = {name:'o4'};
var o4 = Object.create(P); //Object {}
```
